\begin{align*}
\text{Una función modular es una función, } \alpha(q) \text{, que se puede relacionar mediante una} \text{``expresión algebraica''} \text{ llamada ecuación modular con la misma función} \text{ expresada en términos de la misma variable, } q \text{, elevada a una potencia entera: } \alpha(q^p) \text{. La potencia entera, } p \text{, determina el ``orden'' de la ecuación modular.}
\end{align*}
